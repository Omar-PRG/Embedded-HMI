        function [r]= func(ret,app)
              %APPRELEASE=evalin('base','APPRELEASE');
             token = strtok(app.PBCStateLabel.Text,':')
              
             
             if ret==113
                   
                     
                       str= append(token,":Applying");
             disp(str)

                       app.PBCStateLabel.Text=str;
                       
               elseif ret==27
                   
                     
                       str=append (token,":Applied");
                                            disp(str)

                       app.PBCStateLabel.Text=str;

               elseif ret==135
                   
                    
                     str=append (token,":Releasing");
                                          disp(str)

                     app.PBCStateLabel.Text=str;
                     
               elseif ret==54
                   
                      
                     str= append (token,":Released");
                     disp(str)
                      app.PBCStateLabel.Text=str;
               elseif ret==66
                   
                      
                     str= append (token,":Dynamic");
                     disp(str)
                      app.PBCStateLabel.Text=str;
               
               end
               
        end
        
  

    end
